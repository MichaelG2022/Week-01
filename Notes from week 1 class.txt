Class Student {
	public static void main(String[] arg) {

		Student s = new Student();
		s.firstName = "John";
		s.lastName  "Smith";
		s.grade = 80;

		Student t = new Student();
		t.firstName = "Mary";
		t.lastName = "Jones";
		t.grade = 95;

		System.out.println(s.lastName + s.isGoodGrade());
		System.out.println(t.firstName + t.isGoodGrade());

	}
	//Student attributes
	private String firstName;
	private String lastName;
	private int age;
	private int grade;

//method follows - method can be declared anywhere within the Student class
	public boolean isGoodGrade() {	//boolean is return value
		if(grade)>90 return true;
		return false;
}

}
